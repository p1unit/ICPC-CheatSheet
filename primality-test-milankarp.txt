bool miller(ll n,ll d,ll a){
	srand(time(0));
	//ll a=rand()%(n-4) + 2;
	ll x=me(a,d,n);
	if(x==1||x==n-1){
		return true;
	}

	while(d!=n-1){
		//x=(x*x)%n;
		x=mulmod(x,x,n);
		d*=2;
		if(x==1){
			return false;
		}
		if(x==n-1){
			return true;
		}
	}
	return false;
}
